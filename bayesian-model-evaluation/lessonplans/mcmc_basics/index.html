def metropolis_hastings(n_samples, pdf):
    position = 0.
    samples = []
    for _ in range(n_samples):
	proposal = np.random.normal(loc=position)
        accept_prob = pdf(proposal) / pdf(position)
        if np.random.rand() < accept_prob:
            position = proposal
        samples.append(position)
    return samples
